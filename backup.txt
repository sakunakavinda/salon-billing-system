                  .toLowerCase()
                  .includes(filter.toLowerCase()) ||
                sale.customer_tp.includes(filter),
            );
          }

          // NEW: Use API-provided totals instead of calculating client-side
          const totalSalesAmount = salesResult.totalSalesAmount || 0;
          const transactionCount = salesResult.transactionCount || 0;

          // Format currency function
          function formatCurrency(amount) {
            return `Rs. ${amount.toLocaleString()}`;
          }

          // Update the total sales display with API-provided value
          document.getElementById("total-sales-amount").textContent =
            formatCurrency(totalSalesAmount);

          // Update transaction count (this was already working)
          document.getElementById("transaction-count").textContent =
            transactionCount;

          if (filteredSales.length === 0) {
            salesTableBody.innerHTML = `
        <tr>
            <td colspan="4" style="text-align: center; color: var(--text-secondary); padding: 2rem;">
                No sales records found
            </td>
        </tr>
      `;
            return;
          }

          // Render sales data (unchanged)
          filteredSales.forEach((sale) => {
            const row = document.createElement("tr");

            // Get service names for display
            const serviceNames = sale.services.map((service) => {
              const serviceId = service.service_id; // Extract service_id from the object
              const serviceObj = services.find((s) => s.id === serviceId);
              return serviceObj ? serviceObj.service_name : "Unknown Service";
            });

            row.innerHTML = `
        <td data-label="Date & Time">${formatDate(sale.sale_date)}</td>
        <td data-label="Customer">
            <div><strong>${sale.customer_name}</strong></div>
            <div style="font-size: 0.8rem; color: var(--text-secondary);">${sale.customer_tp}</div>
        </td>
        <td data-label="Services">
            <div class="services-list">
                ${serviceNames.map((name) => `<span class="service-tag">${name}</span>`).join("")}
            </div>
        </td>
        <td data-label="Total" class="sale-amount">Rs. ${sale.total_amount.toLocaleString()}</td>
      `;
            salesTableBody.appendChild(row);
          });
        } catch (error) {
          console.error("Error fetching sales history:", error);
          salesTableBody.innerHTML = `
      <tr>
          <td colspan="4" style="text-align: center; color: var(--text-secondary); padding: 2rem;">
              Error loading sales history. Please try again later.
          </td>
      </tr>
    `;
        }
      }

      // Format Date
      function formatDate(dateString) {
        const date = new Date(dateString);
        return date.toLocaleString("en-US", {
          month: "numeric",
          day: "numeric",
          hour: "2-digit",
          minute: "2-digit",
          hour12: false,
        }).replace(',', ' ,');
      }

      // Search Sales
      function searchSales() {
        const searchTerm = searchInput.value.trim();
        const dateFilter = document.getElementById("date-filter").value;

        renderSalesHistory(searchTerm, dateFilter);
      }

      // Clear Search
      function clearSearch() {
        searchInput.value = "";
        document.getElementById("date-filter").value = "today"; // Reset dropdown to default
        renderSalesHistory("", "today");
      }

      // Add event listener for dropdown change
      document
        .getElementById("date-filter")
        .addEventListener("change", function () {
          const searchTerm = searchInput.value.trim();
          const dateFilter = this.value;
          renderSalesHistory(searchTerm, dateFilter);
        });

      // Form Validation
      function validateForm() {
        const customerName = document
          .getElementById("customer-name")
          .value.trim();
        const customerTp = document.getElementById("customer-tp").value.trim();
        const selectedServices = document.querySelectorAll(
          'input[type="checkbox"]:checked',
        );

        let isValid = true;
        let errorMessage = "";

        if (!customerName) {
          errorMessage = "Please enter customer name";
          isValid = false;
        } else if (!customerTp) {
          errorMessage = "Please enter contact number";
          isValid = false;
        } else if (!/^\d{10}$/.test(customerTp)) {
          errorMessage = "Please enter a valid 10-digit contact number";
          isValid = false;
        } else if (selectedServices.length === 0) {
          errorMessage = "Please select at least one service";
          isValid = false;
        }

        if (!isValid) {
          showMessage(errorMessage, "error");
        }

        return isValid;
      }

      // Show Message
      function showMessage(message, type) {
        messageContainer.innerHTML = `<div class="message ${type}">${message}</div>`;
        setTimeout(() => {
          messageContainer.innerHTML = "";
        }, 5000);
      }

      // Show Success Message near Submit Button
      function showSuccessMessage(message) {
        const container = document.getElementById("success-message-container");
        container.innerHTML = `<div class="success-message">${message}</div>`;

        // Auto-hide after 2 seconds
        setTimeout(() => {
          container.innerHTML = "";
        }, 2000);
      }

      // Reset Form
      function resetForm() {
        document.getElementById("customer-name").value = "";
        document.getElementById("customer-tp").value = "";
        document
          .querySelectorAll('input[type="checkbox"]')
          .forEach((checkbox) => {
            checkbox.checked = false;
          });
        calculateTotal();
        showMessage("Form reset successfully", "warning");
      }

      function getCurrentCashierId() {
        return localStorage.getItem("cashierId") || "1";
      }

      // Submit Form
      salesForm.addEventListener("submit", async function (e) {
        e.preventDefault();

        if (!validateForm()) {
          return;
        }

        // Show loading state
        submitBtn.disabled = true;
        submitText.style.display = "none";
        submitLoading.style.display = "inline-block";

        // Get form data
        const customerName = document
          .getElementById("customer-name")
          .value.trim();
        const customerTp = document.getElementById("customer-tp").value.trim();
        const selectedServices = [];
        let totalAmount = 0;

        document
          .querySelectorAll('input[type="checkbox"]:checked')
          .forEach((checkbox) => {
            selectedServices.push(parseInt(checkbox.value));
            totalAmount += parseFloat(checkbox.dataset.price);
          });

        // Replace the setTimeout block with:
        try {
          const response = await fetch("/api/sales", {
            method: "POST",
            headers: {
              "Content-Type": "application/json",
            },
            body: JSON.stringify({
              customerName: customerName,
              customerTp: customerTp,
              services: selectedServices,
              date: new Date().toISOString(),
              totalAmount: totalAmount,
              cashierId: getCurrentCashierId(),
            }),
          });

          const result = await response.json();

          if (result.success) {
            // Success: show success message near submit button for 2 seconds
            showSuccessMessage("Sale recorded successfully!");

            // Reset form after 2 seconds to allow user to see success message
            setTimeout(() => {
              resetForm();
              renderSalesHistory(); // Refresh from database
            }, 2000);
          } else {
            // Error: show error message
            showMessage(result.message || "Failed to save sale", "error");
          }
        } catch (error) {
          console.error("Submit error:", error);
          showMessage("Network error. Please try again.", "error");
        } finally {
          // Reset button state
          submitBtn.disabled = false;
          submitText.style.display = "inline";
          submitLoading.style.display = "none";
        }
      });

      // Logout Function
      function logout() {
        if (confirm("Are you sure you want to logout?")) {
          window.location.href = "/";
        }
      }

      // Add Enter key support for search
      searchInput.addEventListener("keypress", function (e) {
        if (e.key === "Enter") {
          searchSales();
        }
      });
    </script>
